[tex]8\frac{1}{3} = 3 \times 8 + 1 = \frac{25}{3} [/tex]

[tex]2 \frac{7}{8} = 8 \times 2 + 7 = \frac{23}{8} [/tex]

[tex] 2.5 = \frac{25}{10} [/tex]

[tex]12 = \frac{12}{1} [/tex]

[tex] =( \frac{25}{3} \times \frac{8}{23}) + ( \frac{25}{10} \times \frac{12}{1})[/tex]

[tex] = \frac{200}{69} + \frac{300}{10} = \frac{200}{69} \times \frac{10}{300}[/tex]

[tex] = \frac{2000}{690} + \frac{20.700}{690} [/tex]

[tex] = \frac{22.700}{690} = 32.8[/tex]
